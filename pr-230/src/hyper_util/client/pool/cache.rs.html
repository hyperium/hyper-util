1//! A cache of services
2//!
3//! The cache is a single list of cached services, bundled with a `MakeService`.
4//! Calling the cache returns either an existing service, or makes a new one.
5//! The returned `impl Service` can be used to send requests, and when dropped,
6//! it will try to be returned back to the cache.

116    impl<Ev> Builder<Ev> {
117        /// Provide a `Future` executor to be used by the `Cache`.
118        ///
119        /// The executor is used handle some optional background tasks that
120        /// can improve the behavior of the cache, such as reducing connection
121        /// thrashing when a race is won. If not configured with an executor,
122        /// the default behavior is to ignore any of these optional background
123        /// tasks.
124        ///
125        /// The executor should implmenent [`hyper::rt::Executor`].
126        ///
127        /// # Example
128        ///
129        /// ```rust
130        /// # #[cfg(feature = "tokio")]
131        /// # fn run() {
132        /// let _builder = hyper_util::client::pool::cache::builder()
133        ///     .executor(hyper_util::rt::TokioExecutor::new());
134        /// # }
135        /// ```
136        pub fn executor<E>(self, exec: E) -> Builder<events::WithExecutor<E>> {
137            Builder {
138                events: events::WithExecutor(exec),
139            }
140        }

142        /// Build a `Cache` pool around the `connector`.
143        pub fn build<M, Dst>(self, connector: M) -> Cache<M, Dst, Ev>
144        where
145            M: Service<Dst>,
146        {
147            Cache {
148                connector,
149                events: self.events,
150                shared: Arc::new(Mutex::new(Shared {
151                    services: Vec::new(),
152                    waiters: Vec::new(),
153                })),
154            }
155        }
156    }

160    impl<M, Dst, Ev> Service<Dst> for Cache<M, Dst, Ev>
161    where
162        M: Service<Dst>,
163        M::Future: Unpin,
164        M::Response: Unpin,
165        Ev: events::Events<BackgroundConnect<M::Future, M::Response>> + Clone + Unpin,
166    {
167        type Response = Cached<M::Response>;
168        type Error = M::Error;
169        type Future = CacheFuture<M, Dst, Ev>;
170
171        fn poll_ready(&mut self, cx: &mut task::Context<'_>) -> Poll<Result<(), Self::Error>> {
172            if !self.shared.lock().unwrap().services.is_empty() {
173                Poll::Ready(Ok(()))
174            } else {
175                self.connector.poll_ready(cx)
176            }
177        }
178
179        fn call(&mut self, target: Dst) -> Self::Future {
180            // 1. If already cached, easy!
181            let waiter = {
182                let mut locked = self.shared.lock().unwrap();
183                if let Some(found) = locked.take() {
184                    return CacheFuture::Cached {
185                        svc: Some(Cached::new(found, Arc::downgrade(&self.shared))),
186                    };
187                }
188
189                let (tx, rx) = oneshot::channel();
190                locked.waiters.push(tx);
191                rx
192            };
193
194            // 2. Otherwise, we start a new connect, and also listen for
195            //    any newly idle.
196            CacheFuture::Racing {
197                shared: self.shared.clone(),
198                select: future::select(waiter, self.connector.call(target)),
199                events: self.events.clone(),
200            }
201        }
202    }

218    impl<M, Dst, Ev> Future for CacheFuture<M, Dst, Ev>
219    where
220        M: Service<Dst>,
221        M::Future: Unpin,
222        M::Response: Unpin,
223        Ev: events::Events<BackgroundConnect<M::Future, M::Response>> + Unpin,
224    {
225        type Output = Result<Cached<M::Response>, M::Error>;
226
227        fn poll(mut self: Pin<&mut Self>, cx: &mut task::Context<'_>) -> Poll<Self::Output> {
228            loop {
229                match &mut *self.as_mut() {
230                    CacheFuture::Racing {
231                        shared,
232                        select,
233                        events,
234                    } => {
235                        match ready!(Pin::new(select).poll(cx)) {
236                            future::Either::Left((Err(_pool_closed), connecting)) => {
237                                // pool was dropped, so we'll never get it from a waiter,
238                                // but if this future still exists, then the user still
239                                // wants a connection. just wait for the connecting
240                                *self = CacheFuture::Connecting {
241                                    shared: shared.clone(),
242                                    future: connecting,
243                                };
244                            }
245                            future::Either::Left((Ok(pool_got), connecting)) => {
246                                events.on_race_lost(BackgroundConnect {
247                                    future: connecting,
248                                    shared: Arc::downgrade(&shared),
249                                });
250                                return Poll::Ready(Ok(Cached::new(
251                                    pool_got,
252                                    Arc::downgrade(&shared),
253                                )));
254                            }
255                            future::Either::Right((connected, _waiter)) => {
256                                let inner = connected?;
257                                return Poll::Ready(Ok(Cached::new(
258                                    inner,
259                                    Arc::downgrade(&shared),
260                                )));
261                            }
262                        }
263                    }
264                    CacheFuture::Connecting { shared, future } => {
265                        let inner = ready!(Pin::new(future).poll(cx))?;
266                        return Poll::Ready(Ok(Cached::new(inner, Arc::downgrade(&shared))));
267                    }
268                    CacheFuture::Cached { svc } => {
269                        return Poll::Ready(Ok(svc.take().unwrap()));
270                    }
271                }
272            }
273        }
274    }

288    where
289        S: Service<Req>,

292        type Error = S::Error;
293        type Future = S::Future;
294
295        fn poll_ready(&mut self, cx: &mut task::Context<'_>) -> Poll<Result<(), Self::Error>> {
296            self.inner.as_mut().unwrap().poll_ready(cx)

300            self.inner.as_mut().unwrap().call(req)

304    impl<S> Drop for Cached<S> {
305        fn drop(&mut self) {
306            if let Some(value) = self.inner.take() {
307                if let Some(shared) = self.shared.upgrade() {
308                    if let Ok(mut shared) = shared.lock() {
309                        shared.put(value);
310                    }
311                }
312            }
313        }
314    }

316    impl<S: fmt::Debug> fmt::Debug for Cached<S> {

318            f.debug_tuple("Cached")
319                .field(self.inner.as_ref().unwrap())
320                .finish()

326    impl<V> Shared<V> {
327        fn put(&mut self, val: V) {
328            let mut val = Some(val);
329            while let Some(tx) = self.waiters.pop() {
330                if !tx.is_closed() {
331                    match tx.send(val.take().unwrap()) {
332                        Ok(()) => break,
333                        Err(v) => {
334                            val = Some(v);
335                        }
336                    }
337                }
338            }
339
340            if let Some(val) = val {
341                self.services.push(val);
342            }
343        }
344
345        fn take(&mut self) -> Option<V> {
346            // TODO: take in a loop
347            self.services.pop()
348        }
349    }

351    pub struct BackgroundConnect<CF, S> {
352        future: CF,
353        shared: Weak<Mutex<Shared<S>>>,
354    }
355
356    impl<CF, S, E> Future for BackgroundConnect<CF, S>
357    where
358        CF: Future<Output = Result<S, E>> + Unpin,
359    {
360        type Output = ();
361
362        fn poll(mut self: Pin<&mut Self>, cx: &mut task::Context<'_>) -> Poll<Self::Output> {
363            match ready!(Pin::new(&mut self.future).poll(cx)) {
364                Ok(svc) => {
365                    if let Some(shared) = self.shared.upgrade() {
366                        if let Ok(mut locked) = shared.lock() {
367                            locked.put(svc);
368                        }
369                    }
370                    Poll::Ready(())
371                }
372                Err(_e) => Poll::Ready(()),
373            }
374        }
375    }

380    #[non_exhaustive]
381    pub struct Ignore;

383    #[derive(Clone, Debug)]
384    pub struct WithExecutor<E>(pub(super) E);

386    pub trait Events<CF> {
387        fn on_race_lost(&self, fut: CF);
388    }
389
390    impl<CF> Events<CF> for Ignore {
391        fn on_race_lost(&self, _fut: CF) {}
392    }
393
394    impl<E, CF> Events<CF> for WithExecutor<E>
395    where
396        E: hyper::rt::Executor<CF>,
397    {
398        fn on_race_lost(&self, fut: CF) {
399            self.0.execute(fut);
400        }
